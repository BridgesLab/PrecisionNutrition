# Get all coloc objects in your environment
coloc_objects <- ls(pattern = "^coloc\\.")

# Extract summaries into a data frame
coloc_summary <- map_dfr(coloc_objects, function(obj_name) {
  # Get the object
  obj <- get(obj_name)
  
  # Extract gene name and diet suffix from object name (after "coloc.")
  name_parts <- str_remove(obj_name, "^coloc\\.") %>%
    str_to_lower()
  
  # Check if it ends with .hfd or .ncd
  if (str_detect(name_parts, "\\.hfd$")) {
    gene_name <- str_remove(name_parts, "\\.hfd$") %>% str_to_title()
    diet <- "HFD"
  } else if (str_detect(name_parts, "\\.ncd$")) {
    gene_name <- str_remove(name_parts, "\\.ncd$") %>% str_to_title()
    diet <- "NCD"
  } else {
    gene_name <- str_to_title(name_parts)
    diet <- "Additive"
  }
  
  # Get summary - handle different formats
  summary_data <- obj$summary
  
  # Convert to data frame depending on structure
  if (is.matrix(summary_data)) {
    summary_df <- as.data.frame(t(summary_data))
  } else if (is.vector(summary_data) || is.numeric(summary_data)) {
    summary_df <- as.data.frame(t(summary_data))
  } else if (is.data.frame(summary_data)) {
    summary_df <- summary_data
  } else {
    # If unknown format, try to coerce
    summary_df <- as.data.frame(as.list(summary_data))
  }
  
  # Add gene name and diet as first columns
  summary_df %>%
    mutate(Gene = gene_name, Diet = diet, .before = 1)
})
